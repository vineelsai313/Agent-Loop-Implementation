% Heuristics
% Note: intersection is a static, that is why we do not need the time step I or J
% Do not drive through the same intersection in consecutive steps
% because that would require to drive backwards
:- occurs(drive(L1, L2, _), I),
   occurs(drive(L2, L3, _), J),
   next(I, J),
   holds(intersection(L1, L2)),
   holds(intersection(L2, L3)).

% Do not drive through the straight segments in consecutive steps
% because that would require to drive backwards
:- occurs(drive(L1, L2, _), I),
   occurs(drive(L2, L3, _), J),
   next(I, J),
   not holds(intersection(L1, L2)),
   not holds(intersection(L2, L3)).
   
agent_occurs(A, I) :- occurs(A, I), not hpd(A, true, I), agent_action(A).

% --------------------------
% Fluents
% --------------------------
fluent(inertial, has_speed(S)) :- speed(S).
fluent(inertial, at_loc(L)) :- location(L).

fluent(inertial, school_bus_is_stopped(L1, L2)) :- location(L1), location(L2), L1 != L2.
fluent(inertial, pedestrians_are_crossing(L)) :- location(L).
fluent(inertial, light_color(C, L1, L2)) :- color(C), location(L1), location(L2), L1 != L2.

fluent(F) :- fluent(inertial, F).
fluent(F) :- fluent(defined, F).

% --------------------------
% Actions
% --------------------------

% Agent actions:

agent_action(drive(L1, L2, S)) :- location(L1), location(L2), L1!=L2, speed(S), S > 0.
agent_action(stop(L)) :- location(L).


% Non-agent actions:

exogenous_action(school_bus_stops(L1, L2)) :- location(L1), location(L2), L1 != L2.  % A school bus stops between locations L1 and L2
exogenous_action(school_bus_reinitiates(L1, L2)) :- location(L1), location(L2), L1 != L2. % The school bus that is stopped between locations L1 and L2 reinitiates its journey
exogenous_action(pedestrians_start_crossing(L)) :- location(L). % Pedestrians start crossing the street at location L.
exogenous_action(pedestrians_finish_crossing(L)) :- location(L). % Pedestrians finish crossing the street at location L.
exogenous_action(light_turns(C, L1, L2)) :- color(C), location(L1), location(L2), L1 != L2. % The light turns color C between locations L1 and L2 (i.e., at location L1 for the direction towards location L2)

action(A) :- agent_action(A).
action(A) :- exogenous_action(A).

% --------------------------
% Axioms
% --------------------------

% Dynamic causal laws: direct effects of actions

%if an agent is driving to a specific destination it should be at the end.
holds(at_loc(L2), J) :- occurs(drive(L1, L2, S), I),
                        next(I, J),
                        I < n.
%drive changes the has_speed
holds(has_speed(S), J) :- occurs(drive(L1, L2, S), I), 
                          next(I, J),
                          I < n.
%stop changes the has_speed to 0
holds(has_speed(0),J) :- occurs(stop(L),I),
                         next(I, J),
                         I < n.


% State constraints
%If an agent is at a specific location it cannot be at any other location.
-holds(at_loc(L1), I) :- holds(at_loc(L2), I),
                         location(L1),
                         L1 != L2.
%has_speed should be unique
-holds(has_speed(S), I) :- holds(has_speed(S1), I), 
                           speed(S),
                           S != S1.

% Executability conditions
%it is impossible to drive from a to b if the agent is not located at a
-occurs(drive(L1, L2, S), I) :-  -holds(at_loc(L1), I), 
                                 speed(S),
                                 location(L2).
           

% an agent cannot go from point a to b if a and b are not connected
% Note: connected is a static, that is why it does not have the time step I
-occurs(drive(L1, L2, S), I) :- -holds(connected(L1, L2)), 
                                speed(S),
                                step(I).
                                                  
-occurs(stop(L), I) :- -holds(at_loc(L), I).

% Note: intersection is a static, that is why it does not have the time step I
-occurs(drive(L1, L2, S), I) :- holds(intersection(L1, L2)),
                                speed(S),
                                S > 50,
                                step(I).

% --------------------------
% Axioms for non-agent actions:
% --------------------------

holds(school_bus_is_stopped(L1, L2), J) :-
	occurs(school_bus_stops(L1, L2), I),
        next(I, J),
        I < n.

-holds(school_bus_is_stopped(L1, L2), J) :-
	occurs(school_bus_reinitiates(L1, L2), I),
        next(I, J),
        I < n.

holds(pedestrians_are_crossing(L), J) :-
	occurs(pedestrians_start_crossing(L), I),
        next(I, J),
        I < n.

-holds(pedestrians_are_crossing(L), J) :-
	occurs(pedestrians_finish_crossing(L), I),
        next(I, J),
        I < n.

holds(light_color(C, L1, L2), J) :-
	occurs(light_turns(C, L1, L2), I),
        next(I, J),
        I < n.


-holds(light_color(C1, L1, L2), I) :-
	holds(light_color(C2, L1, L2), I),
	color(C1),
	C1 != C2.


% We assume there is only one bus
-occurs(school_bus_stops(L1, L2), I) :-
	holds(school_bus_is_stopped(L1, L2), I).

-occurs(school_bus_reinitiates(L1, L2), I) :-
	-holds(school_bus_is_stopped(L1, L2), I).

-occurs(pedestrians_start_crossing(L), I) :-
	holds(pedestrians_are_crossing(L), I).

-occurs(pedestrians_finish_crossing(L), I) :-
	-holds(pedestrians_are_crossing(L), I).

-occurs(light_turns(C, L1, L2), I) :-
	holds(light_color(C, L1, L2), I).


% --------------------------
%% GENERAL INERTIA AXIOMS FOR INERTIAL FLUENTS:

holds(F,I1) :- fluent(inertial,F),
	holds(F,I),                  
	not -holds(F,I1),
	I < n,
	next(I, I1).

-holds(F,I1) :- fluent(inertial,F),
	-holds(F,I),                
	not holds(F,I1),
	I < n,
	next(I, I1).

% GENERAL CWA FOR DEFINED FLUENTS
-holds(F, I) :- fluent(defined, F),
	not holds(F, I),
	step(I).
